    [docs]    def execute(self, *inputs : Tuple[bytes]) -> bytes:
        pose_list = []
        for c in inputs:
            pose_list += readers.poses(c, self.protobufs)
        nmsed_poses = poses.nms(pose_list, self.height * 0.2)
        return writers.poses(nmsed_poses, self.protobufs)
